14@registry.architectures("spacy.TransitionBasedParser.v2") 

15def build_tb_parser_model( 

16 tok2vec: Model[List[Doc], List[Floats2d]], 

17 state_type: Literal["parser", "ner"], 

18 extra_state_tokens: bool, 

19 hidden_width: int, 

20 maxout_pieces: int, 

21 use_upper: bool, 

22 nO: Optional[int] = None, 

23) -> Model: 

24 """ 

25 Build a transition-based parser model. Can apply to NER or dependency-parsing. 

26 

27 Transition-based parsing is an approach to structured prediction where the 

28 task of predicting the structure is mapped to a series of state transitions. 

29 You might find this tutorial helpful as background: 

30 https://explosion.ai/blog/parsing-english-in-python 

31 

32 The neural network state prediction model consists of either two or three 

33 subnetworks: 

34 

35 * tok2vec: Map each token into a vector representations. This subnetwork 

36 is run once for each batch. 

37 * lower: Construct a feature-specific vector for each (token, feature) pair. 

38 This is also run once for each batch. Constructing the state 

39 representation is then simply a matter of summing the component features 

40 and applying the non-linearity. 

41 * upper (optional): A feed-forward network that predicts scores from the 

42 state representation. If not present, the output from the lower model is 

43 used as action scores directly. 

44 

45 tok2vec (Model[List[Doc], List[Floats2d]]): 

46 Subnetwork to map tokens into vector representations. 

47 state_type (str): 

48 String value denoting the type of parser model: "parser" or "ner" 

49 extra_state_tokens (bool): Whether or not to use additional tokens in the context 

50 to construct the state vector. Defaults to `False`, which means 3 and 8 

51 for the NER and parser respectively. When set to `True`, this would become 6 

52 feature sets (for the NER) or 13 (for the parser). 

53 hidden_width (int): The width of the hidden layer. 

54 maxout_pieces (int): How many pieces to use in the state prediction layer. 

55 Recommended values are 1, 2 or 3. If 1, the maxout non-linearity 

56 is replaced with a ReLu non-linearity if use_upper=True, and no 

57 non-linearity if use_upper=False. 

58 use_upper (bool): Whether to use an additional hidden layer after the state 

59 vector in order to predict the action scores. It is recommended to set 

60 this to False for large pretrained models such as transformers, and True 

61 for smaller networks. The upper layer is computed on CPU, which becomes 

62 a bottleneck on larger GPU-based models, where it's also less necessary. 

63 nO (int or None): The number of actions the model will predict between. 

64 Usually inferred from data at the beginning of training, or loaded from 

65 disk. 

66 """ 

67 if state_type == "parser": 

68 nr_feature_tokens = 13 if extra_state_tokens else 8 

69 elif state_type == "ner": 

70 nr_feature_tokens = 6 if extra_state_tokens else 3 

71 else: 

72 raise ValueError(Errors.E917.format(value=state_type)) 

73 t2v_width = tok2vec.get_dim("nO") if tok2vec.has_dim("nO") else None 

74 tok2vec = chain( 

75 tok2vec, 

76 list2array(), 

77 Linear(hidden_width, t2v_width), 

78 ) 

79 tok2vec.set_dim("nO", hidden_width) 

80 lower = _define_lower( 

81 nO=hidden_width if use_upper else nO, 

82 nF=nr_feature_tokens, 

83 nI=tok2vec.get_dim("nO"), 

84 nP=maxout_pieces, 

85 ) 

86 upper = None 

87 if use_upper: 

88 with use_ops("cpu"): 

89 # Initialize weights at zero, as it's a classification layer. 

90 upper = _define_upper(nO=nO, nI=None) 

91 return TransitionModel(tok2vec, lower, upper, resize_output)

94def _define_upper(nO, nI): 

95 return Linear(nO=nO, nI=nI, init_W=zero_init)

98def _define_lower(nO, nF, nI, nP): 

99 return PrecomputableAffine(nO=nO, nF=nF, nI=nI, nP=nP)

102def resize_output(model, new_nO): 

103 if model.attrs["has_upper"]: 

104 return _resize_upper(model, new_nO) 

105 return _resize_lower(model, new_nO)

108def _resize_upper(model, new_nO): 

109 upper = model.get_ref("upper") 

110 if upper.has_dim("nO") is None: 

111 upper.set_dim("nO", new_nO) 

112 return model 

113 elif new_nO == upper.get_dim("nO"): 

114 return model 

115 

116 smaller = upper 

117 nI = smaller.maybe_get_dim("nI") 

118 with use_ops("cpu"): 

119 larger = _define_upper(nO=new_nO, nI=nI) 

120 # it could be that the model is not initialized yet, then skip this bit 

121 if smaller.has_param("W"): 

122 larger_W = larger.ops.alloc2f(new_nO, nI) 

123 larger_b = larger.ops.alloc1f(new_nO) 

124 smaller_W = smaller.get_param("W") 

125 smaller_b = smaller.get_param("b") 

126 # Weights are stored in (nr_out, nr_in) format, so we're basically 

127 # just adding rows here. 

128 if smaller.has_dim("nO"): 

129 old_nO = smaller.get_dim("nO") 

130 larger_W[:old_nO] = smaller_W 

131 larger_b[:old_nO] = smaller_b 

132 for i in range(old_nO, new_nO): 

133 model.attrs["unseen_classes"].add(i) 

134 

135 larger.set_param("W", larger_W) 

136 larger.set_param("b", larger_b) 

137 model._layers[-1] = larger 

138 model.set_ref("upper", larger) 

139 return model

142def _resize_lower(model, new_nO): 

143 lower = model.get_ref("lower") 

144 if lower.has_dim("nO") is None: 

145 lower.set_dim("nO", new_nO) 

146 return model 

147 

148 smaller = lower 

149 nI = smaller.maybe_get_dim("nI") 

150 nF = smaller.maybe_get_dim("nF") 

151 nP = smaller.maybe_get_dim("nP") 

152 larger = _define_lower(nO=new_nO, nI=nI, nF=nF, nP=nP) 

153 # it could be that the model is not initialized yet, then skip this bit 

154 if smaller.has_param("W"): 

155 larger_W = larger.ops.alloc4f(nF, new_nO, nP, nI) 

156 larger_b = larger.ops.alloc2f(new_nO, nP) 

157 larger_pad = larger.ops.alloc4f(1, nF, new_nO, nP) 

158 smaller_W = smaller.get_param("W") 

159 smaller_b = smaller.get_param("b") 

160 smaller_pad = smaller.get_param("pad") 

161 # Copy the old weights and padding into the new layer 

162 if smaller.has_dim("nO"): 

163 old_nO = smaller.get_dim("nO") 

164 larger_W[:, 0:old_nO, :, :] = smaller_W 

165 larger_pad[:, :, 0:old_nO, :] = smaller_pad 

166 larger_b[0:old_nO, :] = smaller_b 

167 for i in range(old_nO, new_nO): 

168 model.attrs["unseen_classes"].add(i) 

169 

170 larger.set_param("W", larger_W) 

171 larger.set_param("b", larger_b) 

172 larger.set_param("pad", larger_pad) 

173 model._layers[1] = larger 

174 model.set_ref("lower", larger) 

175 return model
