30def _ellipsize(s: str, maxsize: int) -> str: 

31 if len(s) > maxsize: 

32 i = max(0, (maxsize - 3) // 2) 

33 j = max(0, maxsize - 3 - i) 

34 return s[:i] + "..." + s[len(s) - j :] 

35 return s
