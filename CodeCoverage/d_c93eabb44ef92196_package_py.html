64def package( 

65 input_dir: Path, 

66 output_dir: Path, 

67 meta_path: Optional[Path] = None, 

68 code_paths: List[Path] = [], 

69 name: Optional[str] = None, 

70 version: Optional[str] = None, 

71 create_meta: bool = False, 

72 create_sdist: bool = True, 

73 create_wheel: bool = False, 

74 force: bool = False, 

75 silent: bool = True, 

76) -> None: 

77 msg = Printer(no_print=silent, pretty=not silent) 

78 input_path = util.ensure_path(input_dir) 

79 output_path = util.ensure_path(output_dir) 

80 meta_path = util.ensure_path(meta_path) 

81 if create_wheel and not has_wheel(): 

82 err = "Generating a binary .whl file requires wheel to be installed" 

83 msg.fail(err, "pip install wheel", exits=1) 

84 if not input_path or not input_path.exists(): 

85 msg.fail("Can't locate pipeline data", input_path, exits=1) 

86 if not output_path or not output_path.exists(): 

87 msg.fail("Output directory not found", output_path, exits=1) 

88 if create_sdist or create_wheel: 

89 opts = ["sdist" if create_sdist else "", "wheel" if create_wheel else ""] 

90 msg.info(f"Building package artifacts: {', '.join(opt for opt in opts if opt)}") 

91 for code_path in code_paths: 

92 if not code_path.exists(): 

93 msg.fail("Can't find code file", code_path, exits=1) 

94 # Import the code here so it's available when model is loaded (via 

95 # get_meta helper). Also verifies that everything works 

96 util.import_file(code_path.stem, code_path) 

97 if code_paths: 

98 msg.good(f"Including {len(code_paths)} Python module(s) with custom code") 

99 if meta_path and not meta_path.exists(): 

100 msg.fail("Can't find pipeline meta.json", meta_path, exits=1) 

101 meta_path = meta_path or input_dir / "meta.json" 

102 if not meta_path.exists() or not meta_path.is_file(): 

103 msg.fail("Can't load pipeline meta.json", meta_path, exits=1) 

104 meta = srsly.read_json(meta_path) 

105 meta = get_meta(input_dir, meta) 

106 if meta["requirements"]: 

107 msg.good( 

108 f"Including {len(meta['requirements'])} package requirement(s) from " 

109 f"meta and config", 

110 ", ".join(meta["requirements"]), 

111 ) 

112 if name is not None: 

113 if not name.isidentifier(): 

114 msg.fail( 

115 f"Model name ('{name}') is not a valid module name. " 

116 "This is required so it can be imported as a module.", 

117 "We recommend names that use ASCII A-Z, a-z, _ (underscore), " 

118 "and 0-9. " 

119 "For specific details see: https://docs.python.org/3/reference/lexical_analysis.html#identifiers", 

120 exits=1, 

121 ) 

122 if not _is_permitted_package_name(name): 

123 msg.fail( 

124 f"Model name ('{name}') is not a permitted package name. " 

125 "This is required to correctly load the model with spacy.load.", 

126 "We recommend names that use ASCII A-Z, a-z, _ (underscore), " 

127 "and 0-9. " 

128 "For specific details see: https://www.python.org/dev/peps/pep-0426/#name", 

129 exits=1, 

130 ) 

131 meta["name"] = name 

132 if version is not None: 

133 meta["version"] = version 

134 if not create_meta: # only print if user doesn't want to overwrite 

135 msg.good("Loaded meta.json from file", meta_path) 

136 else: 

137 meta = generate_meta(meta, msg) 

138 errors = validate(ModelMetaSchema, meta) 

139 if errors: 

140 msg.fail("Invalid pipeline meta.json") 

141 print("\n".join(errors)) 

142 sys.exit(1) 

143 model_name = meta["name"] 

144 if not model_name.startswith(meta["lang"] + "_"): 

145 model_name = f"{meta['lang']}_{model_name}" 

146 model_name_v = model_name + "-" + meta["version"] 

147 main_path = output_dir / model_name_v 

148 package_path = main_path / model_name 

149 if package_path.exists(): 

150 if force: 

151 shutil.rmtree(str(package_path)) 

152 else: 

153 msg.fail( 

154 "Package directory already exists", 

155 "Please delete the directory and try again, or use the " 

156 "`--force` flag to overwrite existing directories.", 

157 exits=1, 

158 ) 

159 Path.mkdir(package_path, parents=True) 

160 shutil.copytree(str(input_dir), str(package_path / model_name_v)) 

161 for file_name in FILENAMES_DOCS: 

162 file_path = package_path / model_name_v / file_name 

163 if file_path.exists(): 

164 shutil.copy(str(file_path), str(main_path)) 

165 readme_path = main_path / "README.md" 

166 if not readme_path.exists(): 

167 readme = generate_readme(meta) 

168 create_file(readme_path, readme) 

169 create_file(package_path / model_name_v / "README.md", readme) 

170 msg.good("Generated README.md from meta.json") 

171 else: 

172 msg.info("Using existing README.md from pipeline directory") 

173 imports = [] 

174 for code_path in code_paths: 

175 imports.append(code_path.stem) 

176 shutil.copy(str(code_path), str(package_path)) 

177 create_file(main_path / "meta.json", srsly.json_dumps(meta, indent=2)) 

178 create_file(main_path / "setup.py", TEMPLATE_SETUP) 

179 create_file(main_path / "MANIFEST.in", TEMPLATE_MANIFEST) 

180 init_py = TEMPLATE_INIT.format( 

181 imports="\n".join(f"from . import {m}" for m in imports) 

182 ) 

183 create_file(package_path / "__init__.py", init_py) 

184 msg.good(f"Successfully created package directory '{model_name_v}'", main_path) 

185 if create_sdist: 

186 with util.working_dir(main_path): 

187 util.run_command([sys.executable, "setup.py", "sdist"], capture=False) 

188 zip_file = main_path / "dist" / f"{model_name_v}{SDIST_SUFFIX}" 

189 msg.good(f"Successfully created zipped Python package", zip_file) 

190 if create_wheel: 

191 with util.working_dir(main_path): 

192 util.run_command([sys.executable, "setup.py", "bdist_wheel"], capture=False) 

193 wheel_name_squashed = re.sub("_+", "_", model_name_v) 

194 wheel = main_path / "dist" / f"{wheel_name_squashed}{WHEEL_SUFFIX}" 

195 msg.good(f"Successfully created binary wheel", wheel) 

196 if "__" in model_name: 

197 msg.warn( 

198 f"Model name ('{model_name}') contains a run of underscores. " 

199 "Runs of underscores are not significant in installed package names.", 

200 )

212def get_third_party_dependencies( 

213 config: Config, exclude: List[str] = util.SimpleFrozenList() 

214) -> List[str]: 

215 """If the config includes references to registered functions that are 

216 provided by third-party packages (spacy-transformers, other libraries), we 

217 want to include them in meta["requirements"] so that the package specifies 

218 them as dependencies and the user won't have to do it manually. 

219 

220 We do this by: 

221 - traversing the config to check for registered function (@ keys) 

222 - looking up the functions and getting their module 

223 - looking up the module version and generating an appropriate version range 

224 

225 config (Config): The pipeline config. 

226 exclude (list): List of packages to exclude (e.g. that already exist in meta). 

227 RETURNS (list): The versioned requirements. 

228 """ 

229 own_packages = ("spacy", "spacy-legacy", "spacy-nightly", "thinc", "srsly") 

230 distributions = util.packages_distributions() 

231 funcs = defaultdict(set) 

232 # We only want to look at runtime-relevant sections, not [training] or [initialize] 

233 for section in ("nlp", "components"): 

234 for path, value in util.walk_dict(config[section]): 

235 if path[-1].startswith("@"): # collect all function references by registry 

236 funcs[path[-1][1:]].add(value) 

237 for component in config.get("components", {}).values(): 

238 if "factory" in component: 

239 funcs["factories"].add(component["factory"]) 

240 modules = set() 

241 lang = config["nlp"]["lang"] 

242 for reg_name, func_names in funcs.items(): 

243 for func_name in func_names: 

244 # Try the lang-specific version and fall back 

245 try: 

246 func_info = util.registry.find(reg_name, lang + "." + func_name) 

247 except RegistryError: 

248 try: 

249 func_info = util.registry.find(reg_name, func_name) 

250 except RegistryError as regerr: 

251 # lang-specific version being absent is not actually an issue 

252 raise regerr from None 

253 module_name = func_info.get("module") # type: ignore[attr-defined] 

254 if module_name: # the code is part of a module, not a --code file 

255 modules.add(func_info["module"].split(".")[0]) # type: ignore[union-attr] 

256 dependencies = [] 

257 for module_name in modules: 

258 if module_name in distributions: 

259 dist = distributions.get(module_name) 

260 if dist: 

261 pkg = dist[0] 

262 if pkg in own_packages or pkg in exclude: 

263 continue 

264 version = util.get_package_version(pkg) 

265 version_range = util.get_minor_version_range(version) # type: ignore[arg-type] 

266 dependencies.append(f"{pkg}{version_range}") 

267 return dependencies

270def get_build_formats(formats: List[str]) -> Tuple[bool, bool]: 

271 supported = ["sdist", "wheel", "none"] 

272 for form in formats: 

273 if form not in supported: 

274 msg = Printer() 

275 err = f"Unknown build format: {form}. Supported: {', '.join(supported)}" 

276 msg.fail(err, exits=1) 

277 if not formats or "none" in formats: 

278 return (False, False) 

279 return ("sdist" in formats, "wheel" in formats)

287def get_meta( 

288 model_path: Union[str, Path], existing_meta: Dict[str, Any] 

289) -> Dict[str, Any]: 

290 meta: Dict[str, Any] = { 

291 "lang": "en", 

292 "name": "pipeline", 

293 "version": "0.0.0", 

294 "description": "", 

295 "author": "", 

296 "email": "", 

297 "url": "", 

298 "license": "MIT", 

299 } 

300 nlp = util.load_model_from_path(Path(model_path)) 

301 meta.update(nlp.meta) 

302 meta["spacy_version"] = util.get_minor_version_range(about.__version__) 

303 meta.update(existing_meta) 

304 meta["vectors"] = { 

305 "width": nlp.vocab.vectors_length, 

306 "vectors": len(nlp.vocab.vectors), 

307 "keys": nlp.vocab.vectors.n_keys, 

308 "name": nlp.vocab.vectors.name, 

309 } 

310 if about.__title__ != "spacy": 

311 meta["parent_package"] = about.__title__ 

312 meta.setdefault("requirements", []) 

313 # Update the requirements with all third-party packages in the config 

314 existing_reqs = [util.split_requirement(req)[0] for req in meta["requirements"]] 

315 reqs = get_third_party_dependencies(nlp.config, exclude=existing_reqs) 

316 meta["requirements"].extend(reqs) 

317 return meta

343def generate_readme(meta: Dict[str, Any]) -> str: 

344 """ 

345 Generate a Markdown-formatted README text from a model meta.json. Used 

346 within the GitHub release notes and as content for README.md file added 

347 to model packages. 

348 """ 

349 md = MarkdownRenderer() 

350 lang = meta["lang"] 

351 name = f"{lang}_{meta['name']}" 

352 version = meta["version"] 

353 pipeline = ", ".join([md.code(p) for p in meta.get("pipeline", [])]) 

354 components = ", ".join([md.code(p) for p in meta.get("components", [])]) 

355 vecs = meta.get("vectors", {}) 

356 vectors = f"{vecs.get('keys', 0)} keys, {vecs.get('vectors', 0)} unique vectors ({ vecs.get('width', 0)} dimensions)" 

357 author = meta.get("author") or "n/a" 

358 notes = meta.get("notes", "") 

359 license_name = meta.get("license") 

360 sources = _format_sources(meta.get("sources")) 

361 description = meta.get("description") 

362 label_scheme = _format_label_scheme(cast(Dict[str, Any], meta.get("labels"))) 

363 accuracy = _format_accuracy(cast(Dict[str, Any], meta.get("performance"))) 

364 table_data = [ 

365 (md.bold("Name"), md.code(name)), 

366 (md.bold("Version"), md.code(version)), 

367 (md.bold("spaCy"), md.code(meta["spacy_version"])), 

368 (md.bold("Default Pipeline"), pipeline), 

369 (md.bold("Components"), components), 

370 (md.bold("Vectors"), vectors), 

371 (md.bold("Sources"), sources or "n/a"), 

372 (md.bold("License"), md.code(license_name) if license_name else "n/a"), 

373 (md.bold("Author"), md.link(author, meta["url"]) if "url" in meta else author), 

374 ] 

375 # Put together Markdown body 

376 if description: 

377 md.add(description) 

378 md.add(md.table(table_data, ["Feature", "Description"])) 

379 if label_scheme: 

380 md.add(md.title(3, "Label Scheme")) 

381 md.add(label_scheme) 

382 if accuracy: 

383 md.add(md.title(3, "Accuracy")) 

384 md.add(accuracy) 

385 if notes: 

386 md.add(notes) 

387 return md.text

390def _format_sources(data: Any) -> str: 

391 if not data or not isinstance(data, list): 

392 return "n/a" 

393 sources = [] 

394 for source in data: 

395 if not isinstance(source, dict): 

396 source = {"name": source} 

397 name = source.get("name") 

398 if not name: 

399 continue 

400 url = source.get("url") 

401 author = source.get("author") 

402 result = name if not url else "[{}]({})".format(name, url) 

403 if author: 

404 result += " ({})".format(author) 

405 sources.append(result) 

406 return "<br>".join(sources)

409def _format_accuracy(data: Dict[str, Any], exclude: List[str] = ["speed"]) -> str: 

410 if not data: 

411 return "" 

412 md = MarkdownRenderer() 

413 scalars = [(k, v) for k, v in data.items() if isinstance(v, (int, float))] 

414 scores = [ 

415 (md.code(acc.upper()), f"{score*100:.2f}") 

416 for acc, score in scalars 

417 if acc not in exclude 

418 ] 

419 md.add(md.table(scores, ["Type", "Score"])) 

420 return md.text

423def _format_label_scheme(data: Dict[str, Any]) -> str: 

424 if not data: 

425 return "" 

426 md = MarkdownRenderer() 

427 n_labels = 0 

428 n_pipes = 0 

429 label_data = [] 

430 for pipe, labels in data.items(): 

431 if not labels: 

432 continue 

433 col1 = md.bold(md.code(pipe)) 

434 col2 = ", ".join( 

435 [md.code(str(label).replace("|", "\\|")) for label in labels] 

436 ) # noqa: W605 

437 label_data.append((col1, col2)) 

438 n_labels += len(labels) 

439 n_pipes += 1 

440 if not label_data: 

441 return "" 

442 label_info = f"View label scheme ({n_labels} labels for {n_pipes} components)" 

443 md.add("<details>") 

444 md.add(f"<summary>{label_info}</summary>") 

445 md.add(md.table(label_data, ["Component", "Labels"])) 

446 md.add("</details>") 

447 return md.text

450def _is_permitted_package_name(package_name: str) -> bool: 

451 # regex from: https://www.python.org/dev/peps/pep-0426/#name 

452 permitted_match = re.search( 

453 r"^([A-Z0-9]|[A-Z0-9][A-Z0-9._-]*[A-Z0-9])$", package_name, re.IGNORECASE 

454 ) 

455 return permitted_match is not None
