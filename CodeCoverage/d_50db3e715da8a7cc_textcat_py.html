218 def update( 

219 self, 

220 examples: Iterable[Example], 

221 *, 

222 drop: float = 0.0, 

223 sgd: Optional[Optimizer] = None, 

224 losses: Optional[Dict[str, float]] = None, 

225 ) -> Dict[str, float]: 

226 """Learn from a batch of documents and gold-standard information, 

227 updating the pipe's model. Delegates to predict and get_loss. 

228 

229 examples (Iterable[Example]): A batch of Example objects. 

230 drop (float): The dropout rate. 

231 sgd (thinc.api.Optimizer): The optimizer. 

232 losses (Dict[str, float]): Optional record of the loss during training. 

233 Updated using the component name as the key. 

234 RETURNS (Dict[str, float]): The updated losses dictionary. 

235 

236 DOCS: https://spacy.io/api/textcategorizer#update 

237 """ 

238 if losses is None: 

239 losses = {} 

240 losses.setdefault(self.name, 0.0) 

241 validate_examples(examples, "TextCategorizer.update") 

242 self._validate_categories(examples) 

243 if not any(len(eg.predicted) if eg.predicted else 0 for eg in examples): 

244 # Handle cases where there are no tokens in any docs. 

245 return losses 

246 set_dropout_rate(self.model, drop) 

247 scores, bp_scores = self.model.begin_update([eg.predicted for eg in examples]) 

248 loss, d_scores = self.get_loss(examples, scores) 

249 bp_scores(d_scores) 

250 if sgd is not None: 

251 self.finish_update(sgd) 

252 losses[self.name] += loss 

253 return losses

255 def rehearse( 

256 self, 

257 examples: Iterable[Example], 

258 *, 

259 drop: float = 0.0, 

260 sgd: Optional[Optimizer] = None, 

261 losses: Optional[Dict[str, float]] = None, 

262 ) -> Dict[str, float]: 

263 """Perform a "rehearsal" update from a batch of data. Rehearsal updates 

264 teach the current model to make predictions similar to an initial model, 

265 to try to address the "catastrophic forgetting" problem. This feature is 

266 experimental. 

267 

268 examples (Iterable[Example]): A batch of Example objects. 

269 drop (float): The dropout rate. 

270 sgd (thinc.api.Optimizer): The optimizer. 

271 losses (Dict[str, float]): Optional record of the loss during training. 

272 Updated using the component name as the key. 

273 RETURNS (Dict[str, float]): The updated losses dictionary. 

274 

275 DOCS: https://spacy.io/api/textcategorizer#rehearse 

276 """ 

277 if losses is None: 

278 losses = {} 

279 losses.setdefault(self.name, 0.0) 

280 if self._rehearsal_model is None: 

281 return losses 

282 validate_examples(examples, "TextCategorizer.rehearse") 

283 self._validate_categories(examples) 

284 docs = [eg.predicted for eg in examples] 

285 if not any(len(doc) for doc in docs): 

286 # Handle cases where there are no tokens in any docs. 

287 return losses 

288 set_dropout_rate(self.model, drop) 

289 scores, bp_scores = self.model.begin_update(docs) 

290 target, _ = self._rehearsal_model.begin_update(docs) 

291 gradient = scores - target 

292 bp_scores(gradient) 

293 if sgd is not None: 

294 self.finish_update(sgd) 

295 losses[self.name] += (gradient**2).sum() 

296 return losses

298 def _examples_to_truth( 

299 self, examples: Iterable[Example] 

300 ) -> Tuple[numpy.ndarray, numpy.ndarray]: 

301 nr_examples = len(list(examples)) 

302 truths = numpy.zeros((nr_examples, len(self.labels)), dtype="f") 

303 not_missing = numpy.ones((nr_examples, len(self.labels)), dtype="f") 

304 for i, eg in enumerate(examples): 

305 for j, label in enumerate(self.labels): 

306 if label in eg.reference.cats: 

307 truths[i, j] = eg.reference.cats[label] 

308 elif self.support_missing_values: 

309 not_missing[i, j] = 0.0 

310 truths = self.model.ops.asarray(truths) # type: ignore 

311 return truths, not_missing # type: ignore

313 def get_loss(self, examples: Iterable[Example], scores) -> Tuple[float, float]: 

314 """Find the loss and gradient of loss for the batch of documents and 

315 their predicted scores. 

316 

317 examples (Iterable[Examples]): The batch of examples. 

318 scores: Scores representing the model's predictions. 

319 RETURNS (Tuple[float, float]): The loss and the gradient. 

320 

321 DOCS: https://spacy.io/api/textcategorizer#get_loss 

322 """ 

323 validate_examples(examples, "TextCategorizer.get_loss") 

324 self._validate_categories(examples) 

325 truths, not_missing = self._examples_to_truth(examples) 

326 not_missing = self.model.ops.asarray(not_missing) # type: ignore 

327 d_scores = scores - truths 

328 d_scores *= not_missing 

329 mean_square_error = (d_scores**2).mean() 

330 return float(mean_square_error), d_scores

332 def add_label(self, label: str) -> int: 

333 """Add a new label to the pipe. 

334 

335 label (str): The label to add. 

336 RETURNS (int): 0 if label is already present, otherwise 1. 

337 

338 DOCS: https://spacy.io/api/textcategorizer#add_label 

339 """ 

340 if not isinstance(label, str): 

341 raise ValueError(Errors.E187) 

342 if label in self.labels: 

343 return 0 

344 self._allow_extra_label() 

345 self.cfg["labels"].append(label) # type: ignore[attr-defined] 

346 if self.model and "resize_output" in self.model.attrs: 

347 self.model = self.model.attrs["resize_output"](self.model, len(self.labels)) 

348 self.vocab.strings.add(label) 

349 return 1

351 def initialize( 

352 self, 

353 get_examples: Callable[[], Iterable[Example]], 

354 *, 

355 nlp: Optional[Language] = None, 

356 labels: Optional[Iterable[str]] = None, 

357 positive_label: Optional[str] = None, 

358 ) -> None: 

359 """Initialize the pipe for training, using a representative set 

360 of data examples. 

361 

362 get_examples (Callable[[], Iterable[Example]]): Function that 

363 returns a representative sample of gold-standard Example objects. 

364 nlp (Language): The current nlp object the component is part of. 

365 labels (Optional[Iterable[str]]): The labels to add to the component, typically generated by the 

366 `init labels` command. If no labels are provided, the get_examples 

367 callback is used to extract the labels from the data. 

368 positive_label (Optional[str]): The positive label for a binary task with exclusive classes, 

369 `None` otherwise and by default. 

370 

371 DOCS: https://spacy.io/api/textcategorizer#initialize 

372 """ 

373 validate_get_examples(get_examples, "TextCategorizer.initialize") 

374 self._validate_categories(get_examples()) 

375 if labels is None: 

376 for example in get_examples(): 

377 for cat in example.y.cats: 

378 self.add_label(cat) 

379 else: 

380 for label in labels: 

381 self.add_label(label) 

382 if len(self.labels) < 2: 

383 raise ValueError(Errors.E867) 

384 if positive_label is not None: 

385 if positive_label not in self.labels: 

386 err = Errors.E920.format(pos_label=positive_label, labels=self.labels) 

387 raise ValueError(err) 

388 if len(self.labels) != 2: 

389 err = Errors.E919.format(pos_label=positive_label, labels=self.labels) 

390 raise ValueError(err) 

391 self.cfg["positive_label"] = positive_label 

392 subbatch = list(islice(get_examples(), 10)) 

393 doc_sample = [eg.reference for eg in subbatch] 

394 label_sample, _ = self._examples_to_truth(subbatch) 

395 self._require_labels() 

396 assert len(doc_sample) > 0, Errors.E923.format(name=self.name) 

397 assert len(label_sample) > 0, Errors.E923.format(name=self.name) 

398 self.model.initialize(X=doc_sample, Y=label_sample)

400 def _validate_categories(self, examples: Iterable[Example]): 

401 """Check whether the provided examples all have single-label cats annotations.""" 

402 for ex in examples: 

403 vals = list(ex.reference.cats.values()) 

404 if vals.count(1.0) > 1: 

405 raise ValueError(Errors.E895.format(value=ex.reference.cats)) 

406 for val in vals: 

407 if not (val == 1.0 or val == 0.0): 

408 raise ValueError(Errors.E851.format(val=val))
