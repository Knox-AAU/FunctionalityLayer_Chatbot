99# These tags should not be parsed by the usual parser; they're grandfathered 

100# in from RFC 3066. The 'irregular' ones don't fit the syntax at all; the 

101# 'regular' ones do, but would give meaningless results when parsed. 

102# 

103# These are all lowercased so they can be matched case-insensitively, as the 

104# standard requires.

149def _is_ascii(s): 

150 """ 

151 Determine whether a tag consists of ASCII characters. 

152 """ 

153 # When Python 3.6 support is dropped, we can replace this with str.isascii(). 

154 try: 

155 s.encode('ascii') 

156 return True 

157 except UnicodeEncodeError: 

158 return False

161def normalize_characters(tag): 

162 """ 

163 BCP 47 is case-insensitive, and CLDR's use of it considers underscores 

164 equivalent to hyphens. So here we smash tags into lowercase with hyphens, 

165 so we can make exact comparisons. 

166 

167 >>> normalize_characters('en_US') 

168 'en-us' 

169 >>> normalize_characters('zh-Hant_TW') 

170 'zh-hant-tw' 

171 """ 

172 return tag.lower().replace('_', '-')

175def parse_tag(tag): 

176 """ 

177 Parse the syntax of a language tag, without looking up anything in the 

178 registry, yet. Returns a list of (type, value) tuples indicating what 

179 information will need to be looked up. 

180 """ 

181 if not _is_ascii(tag): 

182 raise LanguageTagError("Language tags must be made of ASCII characters") 

183 

184 tag = normalize_characters(tag) 

185 if tag in EXCEPTIONS: 

186 return [('grandfathered', tag)] 

187 else: 

188 # The first subtag is always either the language code, or 'x' to mark 

189 # the entire tag as private-use. Other subtags are distinguished 

190 # by their length and format, but the language code is distinguished 

191 # by the fact that it is required to come first. 

192 subtags = tag.split('-') 

193 

194 # check all subtags for their shape: 1-8 alphanumeric characters 

195 for subtag in subtags: 

196 if len(subtag) < 1 or len(subtag) > 8 or not subtag.isalnum(): 

197 raise LanguageTagError( 

198 f"Expected 1-8 alphanumeric characters, got {subtag!r}" 

199 ) 

200 

201 if subtags[0] == 'x': 

202 if len(subtags) == 1: 

203 raise LanguageTagError("'x' is not a language tag on its own") 

204 # the entire language tag is private use, but we know that, 

205 # whatever it is, it fills the "language" slot 

206 return [('language', tag)] 

207 elif 2 <= len(subtags[0]) <= 4: 

208 # Language codes should be 2 or 3 letters, but 4-letter codes 

209 # are allowed to parse for legacy Unicode reasons 

210 return [('language', subtags[0])] + parse_subtags(subtags[1:]) 

211 else: 

212 subtag_error(subtags[0], 'a language code')

215def parse_subtags(subtags, expect=EXTLANG): 

216 """ 

217 Parse everything that comes after the language tag: scripts, territories, 

218 variants, and assorted extensions. 

219 """ 

220 # We parse the parts of a language code recursively: each step of 

221 # language code parsing handles one component of the code, recurses 

222 # to handle the rest of the code, and adds what it found onto the 

223 # list of things that were in the rest of the code. 

224 # 

225 # This could just as well have been iterative, but the loops would have 

226 # been convoluted. 

227 # 

228 # So here's the base case. 

229 if not subtags: 

230 return [] 

231 

232 # There's a subtag that comes next. We need to find out what it is. 

233 # 

234 # The primary thing that distinguishes different types of subtags is 

235 # length, but the subtags also come in a specified order. The 'expect' 

236 # parameter keeps track of where we are in that order. expect=TERRITORY, 

237 # for example, means we're expecting a territory code, or anything later 

238 # (because everything but the language is optional). 

239 subtag = subtags[0] 

240 tag_length = len(subtag) 

241 

242 # In the usual case, our goal is to recognize what kind of tag this is, 

243 # and set it in 'tagtype' -- as an integer, so we can compare where it 

244 # should go in order. You can see the enumerated list of tagtypes above, 

245 # where the SUBTAG_TYPES global is defined. 

246 tagtype = None 

247 

248 if tag_length == 1: 

249 # A one-letter subtag introduces an extension, which can itself have 

250 # sub-subtags, so we dispatch to a different function at this point. 

251 # 

252 # We don't need to check anything about the order, because extensions 

253 # necessarily come last. 

254 if subtag.isalnum(): 

255 return parse_extension(subtags) 

256 else: 

257 subtag_error(subtag) 

258 

259 elif tag_length == 2: 

260 if subtag.isalpha(): 

261 # Two-letter alphabetic subtags are territories. These are the only 

262 # two-character subtags after the language. 

263 tagtype = TERRITORY 

264 

265 elif tag_length == 3: 

266 if subtag.isalpha(): 

267 # Three-letter alphabetic subtags are 'extended languages'. 

268 # It's allowed for there to be up to three of them in a row, so we 

269 # need another function to enforce that. Before we dispatch to that 

270 # function, though, we need to check whether we're in the right 

271 # place in order. 

272 if expect <= EXTLANG: 

273 return parse_extlang(subtags) 

274 else: 

275 order_error(subtag, EXTLANG, expect) 

276 elif subtag.isdigit(): 

277 # Three-digit subtags are territories representing broad regions, 

278 # such as Latin America (419). 

279 tagtype = TERRITORY 

280 

281 elif tag_length == 4: 

282 if subtag.isalpha(): 

283 # Four-letter alphabetic subtags are scripts. 

284 tagtype = SCRIPT 

285 elif subtag[0].isdigit(): 

286 # Four-character subtags that start with a digit are variants. 

287 tagtype = VARIANT 

288 

289 else: 

290 # Tags of length 5-8 are variants. 

291 tagtype = VARIANT 

292 

293 # That's the end of the big elif block for figuring out what kind of 

294 # subtag we have based on its length. Now we should do something with that 

295 # kind of subtag. 

296 

297 if tagtype is None: 

298 # We haven't recognized a type of tag. This subtag just doesn't fit the 

299 # standard. 

300 subtag_error(subtag) 

301 

302 elif tagtype < expect: 

303 # We got a tag type that was supposed to appear earlier in the order. 

304 order_error(subtag, tagtype, expect) 

305 

306 else: 

307 # We've recognized a subtag of a particular type. If it's a territory or 

308 # script, we expect the next subtag to be a strictly later type, because 

309 # there can be at most one territory and one script. Otherwise, we expect 

310 # the next subtag to be the type we got or later. 

311 

312 if tagtype in (SCRIPT, TERRITORY): 

313 expect = tagtype + 1 

314 else: 

315 expect = tagtype 

316 

317 # Get the name of this subtag type instead of its integer value. 

318 typename = SUBTAG_TYPES[tagtype] 

319 

320 # Some subtags are conventionally written with capitalization. Apply 

321 # those conventions. 

322 if tagtype == SCRIPT: 

323 subtag = subtag.title() 

324 elif tagtype == TERRITORY: 

325 subtag = subtag.upper() 

326 

327 # Recurse on the remaining subtags. 

328 return [(typename, subtag)] + parse_subtags(subtags[1:], expect)

331def parse_extlang(subtags): 

332 """ 

333 Parse an 'extended language' tag, which consists of 1 to 3 three-letter 

334 language codes. 

335 

336 Extended languages are used for distinguishing dialects/sublanguages 

337 (depending on your view) of macrolanguages such as Arabic, Bahasa Malay, 

338 and Chinese. 

339 

340 It's supposed to also be acceptable to just use the sublanguage as the 

341 primary language code, and your code should know what's a macrolanguage of 

342 what. For example, 'zh-yue' and 'yue' are the same language (Cantonese), 

343 and differ only in whether they explicitly spell out that Cantonese is a 

344 kind of Chinese. 

345 """ 

346 index = 0 

347 parsed = [] 

348 while index < len(subtags) and len(subtags[index]) == 3 and index < 3: 

349 parsed.append(('extlang', subtags[index])) 

350 index += 1 

351 return parsed + parse_subtags(subtags[index:], SCRIPT)

354def parse_extension(subtags): 

355 """ 

356 An extension tag consists of a 'singleton' -- a one-character subtag -- 

357 followed by other subtags. Extension tags are in the BCP 47 syntax, but 

358 their meaning is outside the scope of the standard. 

359 

360 For example, there's the u- extension, which is used for setting Unicode 

361 properties in some context I'm not aware of. 

362 

363 If the singleton is 'x', it's a private use extension, and consumes the 

364 rest of the tag. Otherwise, it stops at the next singleton. 

365 """ 

366 subtag = subtags[0] 

367 if len(subtags) == 1: 

368 raise LanguageTagError(f"The subtag {subtag!r} must be followed by something") 

369 

370 if subtag == 'x': 

371 # Private use. Everything after this is arbitrary codes that we 

372 # can't look up. 

373 return [('private', '-'.join(subtags))] 

374 

375 else: 

376 # Look for the next singleton, if there is one. 

377 boundary = 1 

378 while boundary < len(subtags) and len(subtags[boundary]) != 1: 

379 boundary += 1 

380 

381 if boundary == 1: 

382 raise LanguageTagError( 

383 "Tag extensions may not contain two singletons in a row" 

384 ) 

385 # We've parsed a complete extension subtag. Return to the main 

386 # parse_subtags function, but expect to find nothing but more 

387 # extensions at this point. 

388 return [('extension', '-'.join(subtags[:boundary]))] + parse_subtags( 

389 subtags[boundary:], EXTENSION 

390 )

397def order_error(subtag, got, expected): 

398 """ 

399 Output an error indicating that tags were out of order. 

400 """ 

401 options = SUBTAG_TYPES[expected:] 

402 if len(options) == 1: 

403 expect_str = options[0] 

404 elif len(options) == 2: 

405 expect_str = f'{options[0]} or {options[1]}' 

406 else: 

407 joined = ', '.join(options[:-1]) 

408 last = options[-1] 

409 expect_str = f'{joined}, or {last}' 

410 got_str = SUBTAG_TYPES[got] 

411 raise LanguageTagError( 

412 f"This {got_str} subtag, {subtag!r}, is out of place. Expected {expect_str}." 

413 )
