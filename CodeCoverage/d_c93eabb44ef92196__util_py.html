90def parse_config_overrides( 

91 args: List[str], env_var: Optional[str] = ENV_VARS.CONFIG_OVERRIDES 

92) -> Dict[str, Any]: 

93 """Generate a dictionary of config overrides based on the extra arguments 

94 provided on the CLI, e.g. --training.batch_size to override 

95 "training.batch_size". Arguments without a "." are considered invalid, 

96 since the config only allows top-level sections to exist. 

97 

98 env_vars (Optional[str]): Optional environment variable to read from. 

99 RETURNS (Dict[str, Any]): The parsed dict, keyed by nested config setting. 

100 """ 

101 env_string = os.environ.get(env_var, "") if env_var else "" 

102 env_overrides = _parse_overrides(split_arg_string(env_string)) 

103 cli_overrides = _parse_overrides(args, is_cli=True) 

104 if cli_overrides: 

105 keys = [k for k in cli_overrides if k not in env_overrides] 

106 logger.debug("Config overrides from CLI: %s", keys) 

107 if env_overrides: 

108 logger.debug("Config overrides from env variables: %s", list(env_overrides)) 

109 return {**cli_overrides, **env_overrides}

112def _parse_overrides(args: List[str], is_cli: bool = False) -> Dict[str, Any]: 

113 result = {} 

114 while args: 

115 opt = args.pop(0) 

116 err = f"Invalid config override '{opt}'" 

117 if opt.startswith("--"): # new argument 

118 orig_opt = opt 

119 opt = opt.replace("--", "") 

120 if "." not in opt: 

121 if is_cli: 

122 raise NoSuchOption(orig_opt) 

123 else: 

124 msg.fail(f"{err}: can't override top-level sections", exits=1) 

125 if "=" in opt: # we have --opt=value 

126 opt, value = opt.split("=", 1) 

127 opt = opt.replace("-", "_") 

128 else: 

129 if not args or args[0].startswith("--"): # flag with no value 

130 value = "true" 

131 else: 

132 value = args.pop(0) 

133 result[opt] = _parse_override(value) 

134 else: 

135 msg.fail(f"{err}: name should start with --", exits=1) 

136 return result

139def _parse_override(value: Any) -> Any: 

140 # Just like we do in the config, we're calling json.loads on the 

141 # values. But since they come from the CLI, it'd be unintuitive to 

142 # explicitly mark strings with escaped quotes. So we're working 

143 # around that here by falling back to a string if parsing fails. 

144 # TODO: improve logic to handle simple types like list of strings? 

145 try: 

146 return srsly.json_loads(value) 

147 except ValueError: 

148 return str(value)

158 hint_fill: bool = True, 

159): 

160 """Helper to show custom config validation errors on the CLI. 

161 

162 file_path (str / Path): Optional file path of config file, used in hints. 

163 title (str): Override title of custom formatted error. 

164 desc (str): Override description of custom formatted error. 

165 show_config (bool): Whether to output the config the error refers to. 

166 hint_fill (bool): Show hint about filling config. 

167 """ 

168 try: 

169 yield 

170 except ConfigValidationError as e: 

171 title = title if title is not None else e.title 

172 if e.desc: 

173 desc = f"{e.desc}" if not desc else f"{e.desc}\n\n{desc}" 

174 # Re-generate a new error object with overrides 

175 err = e.from_error(e, title="", desc=desc, show_config=show_config) 

176 msg.fail(title) 

177 print(err.text.strip()) 

178 if hint_fill and "value_error.missing" in err.error_types: 

179 config_path = ( 

180 file_path 

181 if file_path is not None and str(file_path) != "-" 

182 else "config.cfg" 

183 ) 

184 msg.text( 

185 "If your config contains missing values, you can run the 'init " 

186 "fill-config' command to fill in all the defaults, if possible:", 

187 spaced=True, 

188 ) 

189 print(f"{COMMAND} init fill-config {config_path} {config_path} \n") 

190 sys.exit(1) 

191 except InterpolationError as e: 

192 msg.fail("Config validation error", e, exits=1)

208def get_git_version( 

209 error: str = "Could not run 'git'. Make sure it's installed and the executable is available.", 

210) -> Tuple[int, int]: 

211 """Get the version of git and raise an error if calling 'git --version' fails. 

212 error (str): The error message to show. 

213 RETURNS (Tuple[int, int]): The version as a (major, minor) tuple. Returns 

214 (0, 0) if the version couldn't be determined. 

215 """ 

216 try: 

217 ret = run_command("git --version", capture=True) 

218 except: 

219 raise RuntimeError(error) 

220 stdout = ret.stdout.strip() 

221 if not stdout or not stdout.startswith("git version"): 

222 return 0, 0 

223 version = stdout[11:].strip().split(".") 

224 return int(version[0]), int(version[1])

237def string_to_list(value: str, intify: bool = False) -> Union[List[str], List[int]]: 

238 """Parse a comma-separated string to a list and account for various 

239 formatting options. Mostly used to handle CLI arguments that take a list of 

240 comma-separated values. 

241 

242 value (str): The value to parse. 

243 intify (bool): Whether to convert values to ints. 

244 RETURNS (Union[List[str], List[int]]): A list of strings or ints. 

245 """ 

246 if not value: 

247 return [] # type: ignore[return-value] 

248 if value.startswith("[") and value.endswith("]"): 

249 value = value[1:-1] 

250 result = [] 

251 for p in value.split(","): 

252 p = p.strip() 

253 if p.startswith("'") and p.endswith("'"): 

254 p = p[1:-1] 

255 if p.startswith('"') and p.endswith('"'): 

256 p = p[1:-1] 

257 p = p.strip() 

258 if intify: 

259 p = int(p) # type: ignore[assignment] 

260 result.append(p) 

261 return result

279def walk_directory(path: Path, suffix: Optional[str] = None) -> List[Path]: 

280 """Given a directory and a suffix, recursively find all files matching the suffix. 

281 Directories or files with names beginning with a . are ignored, but hidden flags on 

282 filesystems are not checked. 

283 When provided with a suffix `None`, there is no suffix-based filtering.""" 

284 if not path.is_dir(): 

285 return [path] 

286 paths = [path] 

287 locs = [] 

288 seen = set() 

289 for path in paths: 

290 if str(path) in seen: 

291 continue 

292 seen.add(str(path)) 

293 if path.parts[-1].startswith("."): 

294 continue 

295 elif path.is_dir(): 

296 paths.extend(path.iterdir()) 

297 elif suffix is not None and not path.parts[-1].endswith(suffix): 

298 continue 

299 else: 

300 locs.append(path) 

301 # It's good to sort these, in case the ordering messes up cache. 

302 locs.sort() 

303 return locs
