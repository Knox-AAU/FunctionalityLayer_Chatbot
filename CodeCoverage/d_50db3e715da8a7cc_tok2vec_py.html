34class Tok2Vec(TrainablePipe): 

35 """Apply a "token-to-vector" model and set its outputs in the doc.tensor 

36 attribute. This is mostly useful to share a single subnetwork between multiple 

37 components, e.g. to have one embedding and CNN network shared between a 

38 parser, tagger and NER. 

39 

40 In order to use the `Tok2Vec` predictions, subsequent components should use 

41 the `Tok2VecListener` layer as the tok2vec subnetwork of their model. This 

42 layer will read data from the `doc.tensor` attribute during prediction. 

43 During training, the `Tok2Vec` component will save its prediction and backprop 

44 callback for each batch, so that the subsequent components can backpropagate 

45 to the shared weights. This implementation is used because it allows us to 

46 avoid relying on object identity within the models to achieve the parameter 

47 sharing. 

48 """ 

49 

50 def __init__(self, vocab: Vocab, model: Model, name: str = "tok2vec") -> None: 

51 """Initialize a tok2vec component. 

52 

53 vocab (Vocab): The shared vocabulary. 

54 model (thinc.api.Model[List[Doc], List[Floats2d]]): 

55 The Thinc Model powering the pipeline component. It should take 

56 a list of Doc objects as input, and output a list of 2d float arrays. 

57 name (str): The component instance name. 

58 

59 DOCS: https://spacy.io/api/tok2vec#init 

60 """ 

61 self.vocab = vocab 

62 self.model = model 

63 self.name = name 

64 self.listener_map: Dict[str, List["Tok2VecListener"]] = {} 

65 self.cfg: Dict[str, Any] = {}

81 def add_listener(self, listener: "Tok2VecListener", component_name: str) -> None: 

82 """Add a listener for a downstream component. Usually internals.""" 

83 self.listener_map.setdefault(component_name, []) 

84 if listener not in self.listener_map[component_name]: 

85 self.listener_map[component_name].append(listener) 

86 

87 def remove_listener(self, listener: "Tok2VecListener", component_name: str) -> bool: 

88 """Remove a listener for a downstream component. Usually internals.""" 

89 if component_name in self.listener_map: 

90 if listener in self.listener_map[component_name]: 

91 self.listener_map[component_name].remove(listener) 

92 # If no listeners are left, remove entry 

93 if not self.listener_map[component_name]: 

94 del self.listener_map[component_name] 

95 return True 

96 return False

98 def find_listeners(self, component) -> None: 

99 """Walk over a model of a processing component, looking for layers that 

100 are Tok2vecListener subclasses that have an upstream_name that matches 

101 this component. Listeners can also set their upstream_name attribute to 

102 the wildcard string '*' to match any `Tok2Vec`. 

103 

104 You're unlikely to ever need multiple `Tok2Vec` components, so it's 

105 fine to leave your listeners upstream_name on '*'. 

106 """ 

107 names = ("*", self.name) 

108 if isinstance(getattr(component, "model", None), Model): 

109 for node in component.model.walk(): 

110 if isinstance(node, Tok2VecListener) and node.upstream_name in names: 

111 self.add_listener(node, component.name)

113 def predict(self, docs: Iterable[Doc]): 

114 """Apply the pipeline's model to a batch of docs, without modifying them. 

115 Returns a single tensor for a batch of documents. 

116 

117 docs (Iterable[Doc]): The documents to predict. 

118 RETURNS: Vector representations for each token in the documents. 

119 

120 DOCS: https://spacy.io/api/tok2vec#predict 

121 """ 

122 if not any(len(doc) for doc in docs): 

123 # Handle cases where there are no tokens in any docs. 

124 width = self.model.get_dim("nO") 

125 return [self.model.ops.alloc((0, width)) for doc in docs] 

126 tokvecs = self.model.predict(docs) 

127 return tokvecs

129 def set_annotations(self, docs: Sequence[Doc], tokvecses) -> None: 

130 """Modify a batch of documents, using pre-computed scores. 

131 

132 docs (Iterable[Doc]): The documents to modify. 

133 tokvecses: The tensors to set, produced by Tok2Vec.predict.

135 DOCS: https://spacy.io/api/tok2vec#set_annotations

141 def update( 

142 self, 

143 examples: Iterable[Example], 

144 *, 

145 drop: float = 0.0, 

146 sgd: Optional[Optimizer] = None, 

147 losses: Optional[Dict[str, float]] = None, 

148 ): 

149 """Learn from a batch of documents and gold-standard information, 

150 updating the pipe's model. 

151 

152 examples (Iterable[Example]): A batch of Example objects. 

153 drop (float): The dropout rate. 

154 sgd (thinc.api.Optimizer): The optimizer. 

155 losses (Dict[str, float]): Optional record of the loss during training. 

156 Updated using the component name as the key. 

157 RETURNS (Dict[str, float]): The updated losses dictionary. 

158 

159 DOCS: https://spacy.io/api/tok2vec#update 

160 """ 

161 if losses is None: 

162 losses = {} 

163 validate_examples(examples, "Tok2Vec.update") 

164 docs = [eg.predicted for eg in examples] 

165 set_dropout_rate(self.model, drop) 

166 tokvecs, bp_tokvecs = self.model.begin_update(docs) 

167 d_tokvecs = [self.model.ops.alloc2f(*t2v.shape) for t2v in tokvecs] 

168 losses.setdefault(self.name, 0.0) 

169 

170 def accumulate_gradient(one_d_tokvecs): 

171 """Accumulate tok2vec loss and gradient. This is passed as a callback 

172 to all but the last listener. Only the last one does the backprop. 

173 """ 

174 nonlocal d_tokvecs 

175 for i in range(len(one_d_tokvecs)): 

176 d_tokvecs[i] += one_d_tokvecs[i] 

177 losses[self.name] += float((one_d_tokvecs[i] ** 2).sum()) 

178 return [self.model.ops.alloc2f(*t2v.shape) for t2v in tokvecs] 

179 

180 def backprop(one_d_tokvecs): 

181 """Callback to actually do the backprop. Passed to last listener.""" 

182 accumulate_gradient(one_d_tokvecs) 

183 d_docs = bp_tokvecs(d_tokvecs) 

184 if sgd is not None: 

185 self.finish_update(sgd) 

186 return d_docs 

187 

188 batch_id = Tok2VecListener.get_batch_id(docs) 

189 for listener in self.listeners[:-1]: 

190 listener.receive(batch_id, tokvecs, accumulate_gradient) 

191 if self.listeners: 

192 self.listeners[-1].receive(batch_id, tokvecs, backprop) 

193 return losses

224class Tok2VecListener(Model): 

225 """A layer that gets fed its answers from an upstream connection, 

226 for instance from a component earlier in the pipeline. 

227 

228 The Tok2VecListener layer is used as a sublayer within a component such 

229 as a parser, NER or text categorizer. Usually you'll have multiple listeners 

230 connecting to a single upstream Tok2Vec component, that's earlier in the 

231 pipeline. The Tok2VecListener layers act as proxies, passing the predictions 

232 from the Tok2Vec component into downstream components, and communicating 

233 gradients back upstream. 

234 """ 

235 

236 name = "tok2vec-listener" 

237 

238 def __init__(self, upstream_name: str, width: int) -> None: 

239 """ 

240 upstream_name (str): A string to identify the 'upstream' Tok2Vec component 

241 to communicate with. The upstream name should either be the wildcard 

242 string '*', or the name of the `Tok2Vec` component. You'll almost 

243 never have multiple upstream Tok2Vec components, so the wildcard 

244 string will almost always be fine. 

245 width (int): 

246 The width of the vectors produced by the upstream tok2vec component. 

247 """ 

248 Model.__init__(self, name=self.name, forward=forward, dims={"nO": width}) 

249 self.upstream_name = upstream_name 

250 self._batch_id: Optional[int] = None 

251 self._outputs = None 

252 self._backprop = None 

253 

254 @classmethod 

255 def get_batch_id(cls, inputs: Iterable[Doc]) -> int: 

256 """Calculate a content-sensitive hash of the batch of documents, to check 

257 whether the next batch of documents is unexpected. 

258 """ 

259 return sum(sum(token.orth for token in doc) for doc in inputs) 

260 

261 def receive(self, batch_id: int, outputs, backprop) -> None: 

262 """Store a batch of training predictions and a backprop callback. The 

263 predictions and callback are produced by the upstream Tok2Vec component, 

264 and later will be used when the listener's component's model is called. 

265 """ 

266 self._batch_id = batch_id 

267 self._outputs = outputs 

268 self._backprop = backprop 

269 

270 def verify_inputs(self, inputs) -> bool: 

271 """Check that the batch of Doc objects matches the ones we have a 

272 prediction for. 

273 """ 

274 if self._batch_id is None and self._outputs is None: 

275 raise ValueError(Errors.E954) 

276 else: 

277 batch_id = self.get_batch_id(inputs) 

278 if batch_id != self._batch_id: 

279 raise ValueError(Errors.E953.format(id1=batch_id, id2=self._batch_id)) 

280 else: 

281 return True

284def forward(model: Tok2VecListener, inputs, is_train: bool): 

285 """Supply the outputs from the upstream Tok2Vec component.""" 

286 if is_train: 

287 # This might occur during training when the tok2vec layer is frozen / hasn't been updated. 

288 # In that case, it should be set to "annotating" so we can retrieve the embeddings from the doc. 

289 if model._batch_id is None: 

290 outputs = [] 

291 for doc in inputs: 

292 if doc.tensor.size == 0: 

293 raise ValueError(Errors.E203.format(name="tok2vec")) 

294 else: 

295 outputs.append(doc.tensor) 

296 return outputs, _empty_backprop 

297 else: 

298 model.verify_inputs(inputs) 

299 return model._outputs, model._backprop 

300 else: 

301 # This is pretty grim, but it's hard to do better :(. 

302 # It's hard to avoid relying on the doc.tensor attribute, because the 

303 # pipeline components can batch the data differently during prediction. 

304 # That doesn't happen in update, where the nlp object works on batches 

305 # of data. 

306 # When the components batch differently, we don't receive a matching 

307 # prediction from the upstream, so we can't predict. 

308 outputs = [] 

309 width = model.get_dim("nO") 

310 for doc in inputs: 

311 if doc.tensor.size == 0: 

312 # But we do need to do *something* if the tensor hasn't been set. 

313 # The compromise is to at least return data of the right shape, 

314 # so the output is valid. 

315 outputs.append(model.ops.alloc2f(len(doc), width)) 

316 else: 

317 outputs.append(doc.tensor) 

318 return outputs, _empty_backprop
