251def getattr_migration(module: str) -> Callable[[str], Any]: 

252 """Implement PEP 562 for objects that were either moved or removed on the migration 

253 to V2. 

254 

255 Args: 

256 module: The module name. 

257 

258 Returns: 

259 A callable that will raise an error if the object is not found. 

260 """ 

261 # This avoids circular import with errors.py. 

262 from .errors import PydanticImportError 

263 

264 def wrapper(name: str) -> object: 

265 """Raise an error if the object is not found, or warn if it was moved. 

266 

267 In case it was moved, it still returns the object. 

268 

269 Args: 

270 name: The object name. 

271 

272 Returns: 

273 The object. 

274 """ 

275 if name == '__path__': 

276 raise AttributeError(f'module {__name__!r} has no attribute {name!r}') 

277 

278 import_path = f'{module}:{name}' 

279 if import_path in MOVED_IN_V2.keys(): 

280 new_location = MOVED_IN_V2[import_path] 

281 warnings.warn(f'`{import_path}` has been moved to `{new_location}`.') 

282 return import_string(MOVED_IN_V2[import_path]) 

283 if import_path in DEPRECATED_MOVED_IN_V2: 

284 # skip the warning here because a deprecation warning will be raised elsewhere 

285 return import_string(DEPRECATED_MOVED_IN_V2[import_path]) 

286 if import_path in REDIRECT_TO_V1: 

287 new_location = REDIRECT_TO_V1[import_path] 

288 warnings.warn( 

289 f'`{import_path}` has been removed. We are importing from `{new_location}` instead.' 

290 'See the migration guide for more details: https://docs.pydantic.dev/latest/migration/' 

291 ) 

292 return import_string(REDIRECT_TO_V1[import_path]) 

293 if import_path == 'pydantic:BaseSettings': 

294 raise PydanticImportError( 

295 '`BaseSettings` has been moved to the `pydantic-settings` package. ' 

296 f'See https://docs.pydantic.dev/{version_short()}/migration/#basesettings-has-moved-to-pydantic-settings ' 

297 'for more details.' 

298 ) 

299 if import_path in REMOVED_IN_V2: 

300 raise PydanticImportError(f'`{import_path}` has been removed in V2.') 

301 globals: Dict[str, Any] = sys.modules[module].__dict__ 

302 if name in globals: 

303 return globals[name] 

304 raise AttributeError(f'module {__name__!r} has no attribute {name!r}') 

305 

306 return wrapper
