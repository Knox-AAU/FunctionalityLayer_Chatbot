39def download( 

40 model: str, 

41 direct: bool = False, 

42 sdist: bool = False, 

43 *pip_args, 

44) -> None: 

45 if ( 

46 not (is_package("spacy") or is_package("spacy-nightly")) 

47 and "--no-deps" not in pip_args 

48 ): 

49 msg.warn( 

50 "Skipping pipeline package dependencies and setting `--no-deps`. " 

51 "You don't seem to have the spaCy package itself installed " 

52 "(maybe because you've built from source?), so installing the " 

53 "package dependencies would cause spaCy to be downloaded, which " 

54 "probably isn't what you want. If the pipeline package has other " 

55 "dependencies, you'll have to install them manually." 

56 ) 

57 pip_args = pip_args + ("--no-deps",) 

58 if direct: 

59 components = model.split("-") 

60 model_name = "".join(components[:-1]) 

61 version = components[-1] 

62 else: 

63 model_name = model 

64 if model in OLD_MODEL_SHORTCUTS: 

65 msg.warn( 

66 f"As of spaCy v3.0, shortcuts like '{model}' are deprecated. Please " 

67 f"use the full pipeline package name '{OLD_MODEL_SHORTCUTS[model]}' instead." 

68 ) 

69 model_name = OLD_MODEL_SHORTCUTS[model] 

70 compatibility = get_compatibility() 

71 version = get_version(model_name, compatibility) 

72 

73 filename = get_model_filename(model_name, version, sdist) 

74 

75 download_model(filename, pip_args) 

76 msg.good(

82def get_model_filename(model_name: str, version: str, sdist: bool = False) -> str: 

83 dl_tpl = "{m}-{v}/{m}-{v}{s}" 

84 suffix = SDIST_SUFFIX if sdist else WHEEL_SUFFIX 

85 filename = dl_tpl.format(m=model_name, v=version, s=suffix) 

86 return filename

89def get_compatibility() -> dict: 

90 if is_prerelease_version(about.__version__): 

91 version: Optional[str] = about.__version__ 

92 else: 

93 version = get_minor_version(about.__version__) 

94 r = requests.get(about.__compatibility__) 

95 if r.status_code != 200: 

96 msg.fail( 

97 f"Server error ({r.status_code})", 

98 f"Couldn't fetch compatibility table. Please find a package for your spaCy " 

99 f"installation (v{about.__version__}), and download it manually. " 

100 f"For more details, see the documentation: " 

101 f"https://spacy.io/usage/models", 

102 exits=1, 

103 ) 

104 comp_table = r.json() 

105 comp = comp_table["spacy"] 

106 if version not in comp: 

107 msg.fail(f"No compatible packages found for v{version} of spaCy", exits=1) 

108 return comp[version]
