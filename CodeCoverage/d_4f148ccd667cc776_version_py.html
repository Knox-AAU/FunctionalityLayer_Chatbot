18def version_info() -> str: 

19 """Return complete version information for Pydantic and its dependencies.""" 

20 import platform 

21 import sys 

22 from pathlib import Path 

23 

24 import pydantic_core._pydantic_core as pdc 

25 

26 if sys.version_info >= (3, 8): 

27 import importlib.metadata as importlib_metadata 

28 else: 

29 import importlib_metadata 

30 

31 # get data about packages that are closely related to pydantic, use pydantic or often conflict with pydantic 

32 package_names = { 

33 'email-validator', 

34 'fastapi', 

35 'mypy', 

36 'pydantic-extra-types', 

37 'pydantic-settings', 

38 'pyright', 

39 'typing_extensions', 

40 } 

41 related_packages = [] 

42 

43 for dist in importlib_metadata.distributions(): 

44 name = dist.metadata['Name'] 

45 if name in package_names: 

46 related_packages.append(f'{name}-{dist.version}') 

47 

48 info = { 

49 'pydantic version': VERSION, 

50 'pydantic-core version': pdc.__version__, 

51 'pydantic-core build': getattr(pdc, 'build_info', None) or pdc.build_profile, 

52 'install path': Path(__file__).resolve().parent, 

53 'python version': sys.version, 

54 'platform': platform.platform(), 

55 'related packages': ' '.join(related_packages), 

56 } 

57 return '\n'.join('{:>30} {}'.format(k + ':', str(v).replace('\n', ' ')) for k, v in info.items())

60def parse_mypy_version(version: str) -> Tuple[int, ...]: 

61 """Parse mypy string version to tuple of ints. 

62 

63 This function is included here rather than the mypy plugin file because the mypy plugin file cannot be imported 

64 outside a mypy run. 

65 

66 It parses normal version like `0.930` and dev version 

67 like `0.940+dev.04cac4b5d911c4f9529e6ce86a27b44f28846f5d.dirty`. 

68 

69 Args: 

70 version: The mypy version string. 

71 

72 Returns: 

73 A tuple of ints. e.g. (0, 930). 

74 """ 

75 return tuple(map(int, version.partition('+')[0].split('.')))
