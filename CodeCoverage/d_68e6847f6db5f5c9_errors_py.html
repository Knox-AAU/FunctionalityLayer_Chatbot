31def filter_warning( 

32 action: Literal["default", "error", "ignore", "always", "module", "once"], 

33 error_msg: str, 

34): 

35 """Customize how spaCy should handle a certain warning. 

36 

37 error_msg (str): e.g. "W006", or a full error message 

38 action (str): "default", "error", "ignore", "always", "module" or "once" 

39 """ 

40 warnings.filterwarnings(action, message=_escape_warning_msg(error_msg))

43def _escape_warning_msg(msg): 

44 """To filter with warnings.filterwarnings, the [] brackets need to be escaped""" 

45 return msg.replace("[", "\\[").replace("]", "\\]")
